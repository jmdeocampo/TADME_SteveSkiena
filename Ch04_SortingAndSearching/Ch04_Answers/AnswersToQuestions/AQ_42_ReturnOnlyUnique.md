	function uniq(arr)  
    {  
        if (!arr || !arr.length)  
        {  
            return arr;  
        }  

        let dict = { };  
        for (let val of arr)  
        {  
            dict[val] = val;  
        }  

        return Object.keys(dict);  
    }  

    uniq([1,1,2,3,5]);
